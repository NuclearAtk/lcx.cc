    Wscript.Echo x(i) & "    " & d & "    " & Chr(d)
    'Wscript.Echo Right("00" & Hex(d), 2)
Next

'-------------------------------------------------

Private Function Bin2Dec(ByVal Bin)
    '二进制字符串转为十进制数字
    Dim i
    For i = 1 To Len(Bin)
        Bin2Dec = Bin2Dec * 2 + CLng(Mid(Bin, i, 1))
    Next
End Function
