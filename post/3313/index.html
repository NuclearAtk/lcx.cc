{

    {
        printf("CreateToolhelp32Snapshot error: %u\n", GetLastError() ); 
        return 0; 
    }
    pe.dwSize = sizeof( PROCESSENTRY32 );

    if ( !Process32FirstW(hSnapshot, &pe) )
    {
        printf("Process32FirstW error: %u\n", GetLastError() ); 
        CloseHandle( hSnapshot );    
        return 0;

    {

{
